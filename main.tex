% header for the resume with name, address, email, phone, website.
\newcommand{\resumeheader}[5]{
    \begin{center}
        \textbf{\LARGE #1} \\
        #2 \\
        #3 $|$ #4 $|$ #5
    \end{center}
    \medskip
}

% big header for sections such as "EDUCATION"
\newcommand{\sectionheader}[1]{
    \noindent\uline{\textbf{\large #1} \hfill}
    \medskip
}

% header for section items such as "___ University"
\newcommand{\sectionitemheader}[4]{
    \noindent\textbf{#1} 
    \hfill
    \textbf{#2} \\
    \noindent\textit{#3}
    \hfill
    #4
}

% header for section items without location/italicized subtitle
\newcommand{\litesectionitemheader}[2]{
    \noindent\textbf{#1} 
    \hfill
    \textbf{#2}
}

% environment for sections such as "EDUCATION"
\newenvironment{bigsection}[1]
    {
        \sectionheader{#1}
            \begin{adjustwidth}{.25cm}{}
    }
    {
            \end{adjustwidth}
            \smallskip
    }

% environment for section with only bullets (no items), such as "TECHNICAL SKILLS"
\newenvironment{litebigsection}[1]
    {
        \sectionheader{#1}
            \begin{adjustwidth}{.25cm}{}
                \begin{itemize}
    }
    {
                \end{itemize}
            \end{adjustwidth}
        \smallskip
    }

% environment for section items such as "___ University"
\newenvironment{sectionitem}[4]
    {
        \sectionitemheader{#1}{#2}{#3}{#4}
        \begin{adjustwidth}{.25cm}{}
            \begin{itemize}
    }
    {
            \end{itemize}
        \end{adjustwidth}
        \medskip
    }

% environment for section items without location/italicized subtitle
\newenvironment{litesectionitem}[2]
    {
        \litesectionitemheader{#1}{#2}
        \begin{adjustwidth}{.25cm}{}
            \begin{itemize}
    }
    {
            \end{itemize}
        \end{adjustwidth}
        \medskip
    }
